\section{Phase 1: Authentication \& Authorization}

\subsection{User Authentication}

\begin{tcolorbox}[title=\textcolor{accent}{POST} /api/auth/validate-registration]
\textbf{Description:} Validates registration data without creating a user. Use this to check if inputs are valid before proceeding.

\vspace{0.5em}
\textbf{Request Body:}
\begin{itemize}
    \item \texttt{firstName} (String, Required)
    \item \texttt{email} (String, Required)
    \item \texttt{password} (String, Required)
    \item \texttt{role} (String, Optional) - Allowed: \texttt{patient}, \texttt{doctor}, \texttt{admin}.
    \item \texttt{jobRole} (String, Required if role is doctor)
    \item \texttt{specialization} (String, Required if role is doctor)
\end{itemize}

\vspace{0.5em}
\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "Validation successful"
}
\end{lstlisting}
\end{tcolorbox}

\begin{tcolorbox}[title=\textcolor{accent}{POST} /api/auth/register]
\textbf{Description:} Registers a new user and returns an authentication token.

\vspace{0.5em}
\textbf{Request Body:} Same as \texttt{validate-registration}.

\vspace{0.5em}
\textbf{Response (201 Created):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "Registration successful",
  "user": {
    "userId": "string",
    "username": "string",
    "role": "string",
    "firstName": "string",
    "lastName": "string",
    "emergencyContact": { "name": null, "phone": null }
  },
  "token": "string"
}
\end{lstlisting}
\end{tcolorbox}

\begin{tcolorbox}[title=\textcolor{accent}{POST} /api/auth/login]
\textbf{Description:} Authenticates a user and sets an HTTP-only cookie with the token.

\vspace{0.5em}
\textbf{Request Body:}
\begin{itemize}
    \item \texttt{username} (String, Required) - Can be username or email.
    \item \texttt{password} (String, Required)
    \item \texttt{role} (String, Optional)
\end{itemize}

\vspace{0.5em}
\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "Login successful",
  "user": { ... },
  "token": "string"
}
\end{lstlisting}
\end{tcolorbox}

\begin{tcolorbox}[title=\textcolor{accent}{POST} /api/auth/logout]
\textbf{Description:} Logs out the user, clears authentication cookies, and destroys the session. If the user is a doctor, sets their status to offline.

\vspace{0.5em}
\textbf{Request Body:} None (Requires Auth Token).

\vspace{0.5em}
\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "Logged out successfully"
}
\end{lstlisting}
\end{tcolorbox}

\begin{tcolorbox}[title=\textcolor{accent}{POST} /api/auth/forgot-password]
\textbf{Description:} Sends password reset instructions to the user's email.

\vspace{0.5em}
\textbf{Request Body:}
\begin{itemize}
    \item \texttt{email} (String, Required)
\end{itemize}

\vspace{0.5em}
\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "Password reset instructions have been sent to your email."
}
\end{lstlisting}
\end{tcolorbox}

\begin{tcolorbox}[title=\textcolor{accent}{POST} /api/auth/reset-password]
\textbf{Description:} Resets the user's password using the token received in email.

\vspace{0.5em}
\textbf{Request Body:}
\begin{itemize}
    \item \texttt{token} (String, Required)
    \item \texttt{newPassword} (String, Required)
\end{itemize}

\vspace{0.5em}
\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "Password reset successful. Please login with your new password."
}
\end{lstlisting}
\end{tcolorbox}

\begin{tcolorbox}[title=\textcolor{accent}{GET} /api/auth/profile]
\textbf{Description:} Retrieves the profile of the currently authenticated user.

\vspace{0.5em}
\textbf{Request Body:} None (Requires Auth Token).

\vspace{0.5em}
\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "user": {
    "userId": "string",
    "username": "string",
    "email": "string",
    "firstName": "string",
    "lastName": "string",
    "role": "string",
    "profileCompleted": boolean
  }
}
\end{lstlisting}
\end{tcolorbox}

\begin{tcolorbox}[title=\textcolor{accent}{GET} /api/protected]
\textbf{Description:} A test route to verify if the token is valid and return user data.

\vspace{0.5em}
\textbf{Request Body:} None (Requires Auth Token).

\vspace{0.5em}
\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]

\end{tcolorbox}

\subsection{Google OAuth}

\begin{tcolorbox}[title=\textcolor{accent}{GET} /api/auth/google]
\textbf{Description:} Initiates the Google OAuth login flow.

\vspace{0.5em}
\textbf{Query Parameters:}
\begin{itemize}
    \item \texttt{role} (String, Optional) - Defaults to \texttt{patient}. Intended role for the user.
\end{itemize}

\vspace{0.5em}
\textbf{Response:} Redirects to Google Consent Screen.
\end{tcolorbox}

\begin{tcolorbox}[title=\textcolor{accent}{GET} /api/auth/google/callback]
\textbf{Description:} Callback route handled by Passport.js after Google authentication.

\vspace{0.5em}
\textbf{Response:} Redirects to frontend with success or error parameters.
\end{tcolorbox}

\subsection{OTP Verification}

\begin{tcolorbox}[title=\textcolor{accent}{POST} /api/otp/send]
\textbf{Description:} Sends a 6-digit OTP to the specified email for verification during signup.

\vspace{0.5em}
\textbf{Request Body:}
\begin{itemize}
    \item \texttt{email} (String, Required)
    \item \texttt{userType} (String, Required) - Allowed: \texttt{patient}, \texttt{doctor}.
\end{itemize}

\vspace{0.5em}
\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "OTP sent to your email",
  "expiresAt": "string",
  "email": "string"
}
\end{lstlisting}
\end{tcolorbox}

\begin{tcolorbox}[title=\textcolor{accent}{POST} /api/otp/verify]
\textbf{Description:} Verifies the OTP code sent to the email.

\vspace{0.5em}
\textbf{Request Body:}
\begin{itemize}
    \item \texttt{email} (String, Required)
    \item \texttt{otp} (String, Required) - 6 digit number.
\end{itemize}

\vspace{0.5em}
\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "Email verified successfully",
  "verified": true
}
\end{lstlisting}
\end{tcolorbox}

\begin{tcolorbox}[title=\textcolor{accent}{POST} /api/otp/resend]
\textbf{Description:} Resends the OTP to the email if the previous one expired or was not received.

\vspace{0.5em}
\textbf{Request Body:}
\begin{itemize}
    \item \texttt{email} (String, Required)
\end{itemize}

\vspace{0.5em}
\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "OTP resent to your email",
  "expiresAt": "string",
  "resendsLeft": number
}
\end{lstlisting}
\end{tcolorbox}

    \item \texttt{emergencyContact} (Object)

\textbf{Response (200 OK):}
\begin{lstlisting}[language=json]
{
  "success": true,
  "message": "Profile setup completed successfully",
